101  bool Interpret(string command)
102  {
103  if (mCurCommandIdx == 0) {
104  const auto p1 = std::chrono::system_clock::now();
105  mCurBlock.mTimestamp = std::chrono::duration_cast<std::chrono::seconds>(p1.time_since_epoch()).count();
106  }
107 
108  if (command != "}" && command != "{") mCurCommandIdx++;
109  mCurBlock.mCommands.push_back(command); // does contain brackets
110 
111  if (command == "}") {
112  if (--mEmbedBracers == 0) // that is a dyn block finish
113  return true;
114  return false;
115  }
116 
117  if (command == "{") { // this is an emergency exit for the block
118  mAmDynBlock = true;
119  if ((++mEmbedBracers == 1) && (mCurCommandIdx > 0))
120  return true;
121  return false;
122  }
123 
124  if ((mCurCommandIdx == mN) && (mEmbedBracers == 0)) // block is finished
125  {
126  return true;
127  }
128  return false;
129  }
130 
131  void Refresh()
132  {
133  mCurCommandIdx = 0;
134  mEmbedBracers = 0;
135  if (mCurBlock.mCommands.size() > 0 && mCurBlock.mCommands[mCurBlock.mCommands.size() - 1] == "{") {
136  mEmbedBracers++;
137  mAmDynBlock = true;
138  }
139  mCurBlock.mCommands.clear();
140  return;
141  }
142

146  }
